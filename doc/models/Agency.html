def self.with_role_condition(user)
  scope = all

  case user.role.name.to_sym
  when :town_manager
    scope.joins(:town_managers).where(admins: { id: user.id })
  when :parking_admin
    scope.none
  when :manager
    scope.joins(:managers).where(admins: { id: user.id })
  when :officer
    scope.joins(:officers).where(admins: { id: user.id })
  else
    scope
  end
end
